**Target — stale-dirty policy** (new function signature; call site in `spawn_background_git_refresh` line 338 replaces `update_tray_dirty_icon(&app, s.any_dirty)`):
```rust
pub(crate) fn update_tray_icon_state(app: &AppHandle, repos: &[RepoDto], stale_dirty_days: u32, syncing: bool) {
    let Some(tray) = app.tray_by_id("main") else { return; };
    let Some(icons) = app.try_state::<crate::tray::TrayIcons>() else { return; };

    let icon = if syncing {
        icons.syncing_frame(0).clone()   // first sync frame (see tray.rs TrayIcons extension)
    } else if has_stale_dirty(repos, stale_dirty_days) {
        icons.stale_dirty.clone()
    } else {
        icons.default.clone()
    };
    let _ = tray.set_icon(Some(icon));
}

fn has_stale_dirty(repos: &[RepoDto], stale_dirty_days: u32) -> bool {
    let threshold_secs = stale_dirty_days as i64 * 86_400;
    let now = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .map(|d| d.as_secs() as i64)
        .unwrap_or(0);
    repos.iter().any(|r| {
        r.is_dirty == Some(true) && {
            let age = match r.last_opened_at {
                Some(t) => now - t,
                None => i64::MAX,  // never opened + dirty => treat as stale
            };
            age >= threshold_secs
        }
    })
}
```
